-- ===== PLAYER LIST (INCLUDING SELF) =====
local function UpdatePlayerList(gui, filter)
    filter = filter or ""
    filter = filter:lower()
    
    for _, child in pairs(gui.PlayerList:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end
    
    local yPos = 0
    local playerCount = 0
    
    local sortedPlayers = {}
    for _, plr in pairs(Players:GetPlayers()) do
        table.insert(sortedPlayers, plr)
    end
    table.sort(sortedPlayers, function(a, b)
        return a.Name:lower() < b.Name:lower()
    end)
    
    for _, plr in pairs(sortedPlayers) do
        local name = plr.Name
        local displayName = plr.DisplayName
        local isSelf = (plr == LocalPlayer)
        
        if filter == "" or name:lower():find(filter, 1, true) or displayName:lower():find(filter, 1, true) then
            local btn = Instance.new("TextButton")
            btn.Size = UDim2.new(1, -6, 0, 28)
            btn.Position = UDim2.new(0, 3, 0, yPos)
            btn.BackgroundColor3 = isSelf and Color3.fromRGB(50, 15, 20) or Color3.fromRGB(25, 25, 35)
            btn.BackgroundTransparency = 0.3
            btn.Text = ""
            btn.BorderSizePixel = 0
            btn.Parent = gui.PlayerList
            
            local btnCorner = Instance.new("UICorner")
            btnCorner.CornerRadius = UDim.new(0, 4)
            btnCorner.Parent = btn
            
            local prefix = isSelf and "⭐ " or ""
            local nameLabel = Instance.new("TextLabel")
            nameLabel.Size = UDim2.new(1, -10, 1, 0)
            nameLabel.Position = UDim2.new(0, 8, 0, 0)
            nameLabel.BackgroundTransparency = 1
            nameLabel.Text = prefix .. name .. " (" .. displayName .. ")"
            nameLabel.TextColor3 = isSelf and Color3.fromRGB(255, 200, 100) or Color3.fromRGB(220, 220, 230)
            nameLabel.Font = Enum.Font.Gotham
            nameLabel.TextSize = 11
            nameLabel.TextXAlignment = Enum.TextXAlignment.Left
            nameLabel.Parent = btn
            
            btn.MouseEnter:Connect(function()
                btn.BackgroundColor3 = isSelf and Color3.fromRGB(70, 25, 30) or Color3.fromRGB(45, 45, 60)
            end)
            btn.MouseLeave:Connect(function()
                btn.BackgroundColor3 = isSelf and Color3.fromRGB(50, 15, 20) or Color3.fromRGB(25, 25, 35)
            end)
            
            btn.MouseButton1Click:Connect(function()
                gui.ManualNameBox.Text = name
            end)
            
            yPos = yPos + 30
            playerCount = playerCount + 1
        end
    end
    
    if playerCount == 0 then
        local emptyLabel = Instance.new("TextLabel")
        emptyLabel.Size = UDim2.new(1, 0, 0, 28)
        emptyLabel.BackgroundTransparency = 1
        emptyLabel.Text = filter == "" and "Tidak ada player" or "Tidak ditemukan..."
        emptyLabel.TextColor3 = Color3.fromRGB(120, 120, 130)
        emptyLabel.Font = Enum.Font.Gotham
        emptyLabel.TextSize = 11
        emptyLabel.Parent = gui.PlayerList
    end
    
    gui.PlayerList.CanvasSize = UDim2.new(0, 0, 0, math.max(yPos + 4, 40))
end

-- ===== EXPLOIT ACTIONS =====
local function JailPlayer(username, duration)
    if not username or username == "" then
        ShowNotification("❌ Error", "Masukkan username!", Color3.fromRGB(200, 50, 50))
        return false
    end
    local dur = tonumber(duration) or 60
    if dur < 1 then dur = 1 end
    
    local success, err = pcall(function()
        JailRemote:FireServer("Jail", username, tostring(dur))
    end)
    
    if success then
        ShowNotification("⛓ JAILED", username .. " → " .. dur .. " detik", Color3.fromRGB(200, 30, 30))
        return true
    else
        ShowNotification("❌ Gagal", tostring(err), Color3.fromRGB(200, 50, 50))
