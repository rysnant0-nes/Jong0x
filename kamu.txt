-- Kumpulin remote dengan safety check
local function getRemote(name)
    local r = Events:FindFirstChild(name)
    if not r then
        -- coba case insensitive
        for _, v in pairs(Events:GetChildren()) do
            if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
                if v.Name:lower() == name:lower() then
                    r = v
                    break
                end
            end
        end
    end
    if r then
        print("  [OK] Remote:", r.Name, "(", r.ClassName, ")")
    else
        warn("  [!] Remote tidak ditemukan:", name)
    end
    return r
end

-- Safe fire
local function fire(r, ...)
    if not r then warn("[!] Remote nil, cannot fire") return false end
    local s, e = pcall(function() r:FireServer(...) end)
    if not s then warn("[!] FireServer error:", e) end
    return s
end

local function invoke(r, ...)
    if not r then warn("[!] Remote nil, cannot invoke") return nil end
    local s, result = pcall(function() return r:InvokeServer(...) end)
    if not s then warn("[!] InvokeServer error:", result) end
    return s and result or nil
end
