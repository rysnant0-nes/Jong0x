local lastError = ""

local function SafeFireServer(...)
    local args = {...}
    print("[JAIL EXPLOIT DEBUG] FireServer called with:", unpack(args))
    
    -- Coba tanpa pcall dulu biar error kebaca (tapi dibungkus spawning)
    local success, result = pcall(function()
        JailRemote:FireServer(unpack(args))
    end)
    
    if success then
        print("[JAIL EXPLOIT DEBUG] FireServer SUCCESS")
        return true
    else
        print("[JAIL EXPLOIT DEBUG] FireServer FAILED:", tostring(result))
        lastError = tostring(result)
        return false, result
    end
end

-- ===== PLAYER LIST UPDATER =====
local function UpdatePlayerList(filter)
    filter = (filter or ""):lower()
    playerList:ClearAllChildren()
    
    local yPos = 0
    local count = 0
    
    local sorted = {}
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer then
            table.insert(sorted, plr)
        end
    end
    table.sort(sorted, function(a,b) return a.Name:lower() < b.Name:lower() end)
    
    for _, plr in pairs(sorted) do
        local name = plr.Name
        local display = plr.DisplayName
        
        if filter == "" or name:lower():find(filter,1,true) or display:lower():find(filter,1,true) then
            local btn = Instance.new("TextButton")
            btn.Size = UDim2.new(1, -6, 0, 28)
            btn.Position = UDim2.new(0, 3, 0, yPos)
            btn.BackgroundColor3 = Color3.fromRGB(22, 22, 32)
            btn.BackgroundTransparency = 0.2
            btn.Text = ""
            btn.BorderSizePixel = 0
            btn.Parent = playerList
            
            local bCorner = Instance.new("UICorner")
            bCorner.CornerRadius = UDim.new(0, 4)
            bCorner.Parent = btn
            
            local nLabel = Instance.new("TextLabel")
            nLabel.Size = UDim2.new(1, -10, 1, 0)
            nLabel.Position = UDim2.new(0, 8, 0, 0)
            nLabel.BackgroundTransparency = 1
            nLabel.Text = name .. " (" .. display .. ")"
            nLabel.TextColor3 = Color3.fromRGB(220, 220, 230)
            nLabel.Font = Enum.Font.Gotham
            nLabel.TextSize = 11
            nLabel.TextXAlignment = Enum.TextXAlignment.Left
            nLabel.Parent = btn
            
            btn.MouseEnter:Connect(function()
                btn.BackgroundColor3 = Color3.fromRGB(40, 40, 55)
            end)
            btn.MouseLeave:Connect(function()
                btn.BackgroundColor3 = Color3.fromRGB(22, 22, 32)
            end)
            btn.MouseButton1Click:Connect(function()
                manualNameBox.Text = name
                -- Auto-focus ke jail button
                jailBtn:CaptureFocus()
            end)
            
            yPos = yPos + 30
            count = count + 1
        end
    end
    
    if count == 0 then
        local empty = Instance.new("TextLabel")
        empty.Size = UDim2.new(1, 0, 0, 28)
        empty.BackgroundTransparency = 1
        empty.Text = filter == "" and "Tidak ada player lain" or "Tidak ditemukan"
        empty.TextColor3 = Color3.fromRGB(110, 110, 120)
        empty.Font = Enum.Font.Gotham
        empty.TextSize = 11
        empty.Parent = playerList
        yPos = 30
    end
    
    playerList.CanvasSize = UDim2.new(0, 0, 0, yPos + 4)
end

local function JailPlayer(username, duration)
    if not username or username == "" then
        ShowNotif("❌ ERROR", "Username kosong!", Color3.fromRGB(200, 50, 50), 3)
        return false, "Empty username"
    end
    
    local dur = tonumber(duration)
    if not dur or dur < 1 then
        dur = 60
    end
    
    print("[JAIL] Attempting to jail:", username, "for", dur, "seconds")
    print("[JAIL] Remote:", JailRemote:GetFullName())
    
    local success, err = SafeFireServer("Jail", username, tostring(dur))
    
    if success then
        ShowNotif("⛓ JAILED", username .. " → " .. dur .. "s", Color3.fromRGB(200, 30, 30))
        print("[JAIL] SUCCESS:", username, "jailed for", dur, "s")
        return true
    else
        local errMsg = tostring(err or "Unknown error")
        ShowNotif("❌ FAILED", errMsg, Color3.fromRGB(200, 50, 50), 5)
        print("[JAIL] FAILED:", errMsg)
        return false, err
    end
end

local function ReleasePlayer(username)
    if not username or username == "" then
        ShowNotif("❌ ERROR", "Username kosong!", Color3.fromRGB(200, 50, 50), 3)
        return false, "Empty username"
    end
    
    print("[RELEASE] Attempting to release:", username)
    
    local success, err = SafeFireServer("Release", username)
    
    if success then
        ShowNotif("🔓 RELEASED", username .. " dibebaskan", Color3.fromRGB(30, 150, 50))
        print("[RELEASE] SUCCESS:", username)
        return true
    else
        local errMsg = tostring(err or "Unknown error")
        ShowNotif("❌ FAILED", errMsg, Color3.fromRGB(200, 50, 50), 5)
        print("[RELEASE] FAILED:", errMsg)
        return false, err
    end
end

local function JailAllPlayers(duration)
    local dur = tonumber(duration) or 60
    if dur < 1 then dur = 60 end
    
    local count = 0
    local errors = {}
    
    for _, plr in pairs(Players:GetPlayers()) do
        local ok, err = JailPlayer(plr.Name, dur)
        if ok then
            count = count + 1
        else
            table.insert(errors, plr.Name .. ":" .. tostring(err))
        end
        task.wait(0.15)
    end
    
    if #errors > 0 then
        ShowNotif("⚡ PARTIAL", count .. " di-jail, " .. #errors .. " gagal. Cek F9", Color3.fromRGB(200, 100, 30), 4)
        print("[JAIL ALL] Errors:", unpack(errors))
    else
        ShowNotif("⚡ COMPLETE", count .. " player di-jail", Color3.fromRGB(200, 100, 30))
    end
end

local function ReleaseAllPlayers()
    local count = 0
    local errors = {}
    
    for _, plr in pairs(Players:GetPlayers()) do
        local ok, err = ReleasePlayer(plr.Name)
        if ok then
            count = count + 1
        else
            table.insert(errors, plr.Name .. ":" .. tostring(err))
        end
        task.wait(0.15)
    end
    
    if #errors > 0 then
        ShowNotif("🔄 PARTIAL", count .. " dibebaskan, " .. #errors .. " gagal", Color3.fromRGB(200, 150, 30), 4)
        print("[RELEASE ALL] Errors:", unpack(errors))
    else
        ShowNotif("🔄 COMPLETE", count .. " player dibebaskan", Color3.fromRGB(30, 150, 50))
    end
end
