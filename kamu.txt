-- UTILITY
local function getChar(p) return p and p.Character or nil end
local function getHum(p)
    local c = getChar(p)
    return c and c:FindFirstChild("Humanoid") or nil
end
local function getRoot(p)
    local c = getChar(p)
    return c and c:FindFirstChild("HumanoidRootPart") or nil
end
local function getAllPlayers()
    local list = {}
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and getHum(p) and getHum(p).Health > 0 then
            table.insert(list, p)
        end
    end
    return list
end

-- SAFE FIRE
local function fire(remote, ...)
    if not remote then warn("[!] Remote nil") return false end
    local s, e = pcall(function() remote:FireServer(...) end)
    if not s then warn("[!] FireServer error:", e) end
    return s
end

local function invoke(remote, ...)
    if not remote then warn("[!] Remote nil") return nil end
    local s, r = pcall(function() return remote:InvokeServer(...) end)
    if not s then warn("[!] InvokeServer error:", r) end
    return s and r or nil
end
