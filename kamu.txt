-- ===== FUNGSI EXPLOIT =====
local vehicleCache = {}  -- Simpan daftar kendaraan

local function FetchVehicles()
    print("[CAR] Meminta daftar kendaraan...")
    local success, err = pcall(function()
        CarRemote:FireServer("GetVehicles")
    end)
    if not success then
        ShowNotif("❌ GAGAL", "GetVehicles: " .. tostring(err), false)
    end
end

local function SpawnVehicle(name)
    if not name or name == "" then
        ShowNotif("❌ ERROR", "Nama kendaraan kosong!", false)
        return
    end
    
    print("[CAR] Spawning vehicle:", name)
    local success, err = pcall(function()
        CarRemote:FireServer("SpawnVehicle", name)
    end)
    
    if success then
        ShowNotif("✅ SPAWNED", name:upper() .. " berhasil di-spawn!", true)
    else
        ShowNotif("❌ GAGAL", tostring(err), false)
    end
end

local function DeleteVehicle()
    print("[CAR] Menghapus kendaraan...")
    local success, err = pcall(function()
        CarRemote:FireServer("DeleteVehicle")
    end)
    
    if success then
        ShowNotif("🗑️ DELETED", "Kendaraan dihapus!", true)
    else
        ShowNotif("❌ GAGAL", tostring(err), false)
    end
end

local function SpawnAllVehicles()
    local count = 0
    for _, v in pairs(vehicleCache) do
        local ok, _ = pcall(function()
            CarRemote:FireServer("SpawnVehicle", v)
        end)
        if ok then count = count + 1 end
        task.wait(0.15)
    end
    ShowNotif("⚡ BULK SPAWN", count .. " kendaraan di-spawn!", true)
end

-- Intercept server response
CarRemote.OnClientEvent:Connect(function(action, data1, data2)
    if action == "RenderList" then
        -- Simpan cache
        vehicleCache = {}
        for _, v in pairs(data1) do
            table.insert(vehicleCache, v)
        end
        print("[CAR] Daftar kendaraan diterima:", #vehicleCache, "kendaraan")
        ShowNotif("📋 VEHICLES", #vehicleCache .. " kendaraan tersedia", true)
        
        -- Update GUI vehicle list
        UpdateVehicleList(vehicleCache)
        
    elseif action == "Notification" then
        ShowNotif(data1, data2, data1 == "BERHASIL")
    end
end)

function UpdateVehicleList(vehicles)
    -- Clear existing
    for _, child in pairs(vehicleList:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end
    
    if not vehicles or #vehicles == 0 then
        local empty = Instance.new("TextLabel")
        empty.Size = UDim2.new(1, 0, 0, 30)
        empty.BackgroundTransparency = 1
        empty.Text = "Tidak ada kendaraan. Klik FETCH"
        empty.TextColor3 = Color3.fromRGB(100, 100, 110)
        empty.Font = Enum.Font.Gotham
        empty.TextSize = 11
        empty.Parent = vehicleList
        return
    end
    
    local yPos = 0
    for _, vehicleName in pairs(vehicles) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(1, -6, 0, 30)
        btn.Position = UDim2.new(0, 3, 0, yPos)
        btn.BackgroundColor3 = Color3.fromRGB(22, 22, 32)
        btn.BackgroundTransparency = 0.2
        btn.Text = ""
        btn.BorderSizePixel = 0
        btn.Parent = vehicleList
        
        local bCorner = Instance.new("UICorner")
        bCorner.CornerRadius = UDim.new(0, 4)
        bCorner.Parent = btn
        
        local nameLabel = Instance.new("TextLabel")
        nameLabel.Size = UDim2.new(1, -10, 1, 0)
        nameLabel.Position = UDim2.new(0, 8, 0, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text = "🚙 " .. vehicleName:upper()
        nameLabel.TextColor3 = Color3.fromRGB(220, 220, 230)
        nameLabel.Font = Enum.Font.GothamMedium
        nameLabel.TextSize = 11
        nameLabel.TextXAlignment = Enum.TextXAlignment.Left
        nameLabel.Parent = btn
        
        local spawnIcon = Instance.new("TextLabel")
        spawnIcon.Size = UDim2.new(0, 30, 1, 0)
        spawnIcon.Position = UDim2.new(1, -30, 0, 0)
        spawnIcon.BackgroundTransparency = 1
        spawnIcon.Text = "▶"
        spawnIcon.TextColor3 = Color3.fromRGB(60, 180, 60)
        spawnIcon.Font = Enum.Font.GothamBold
        spawnIcon.TextSize = 14
        spawnIcon.Parent = btn
        
        -- Hover
        btn.MouseEnter:Connect(function()
            btn.BackgroundColor3 = Color3.fromRGB(40, 40, 55)
        end)
        btn.MouseLeave:Connect(function()
            btn.BackgroundColor3 = Color3.fromRGB(22, 22, 32)
        end)
        
        -- Click to spawn
        btn.MouseButton1Click:Connect(function()
            SpawnVehicle(vehicleName)
        end)
        
        yPos = yPos + 32
    end
    
    vehicleList.CanvasSize = UDim2.new(0, 0, 0, yPos + 4)
end
