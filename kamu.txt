    local c = ReplicatedStorage:FindFirstChild("CarbonResource")
    if not c then

            if v:FindFirstChild("Events") and v:FindFirstChild("Global") then
                return v
            end
        end
    end
    return c

-- Utility
local function getChar(p)
    return p and p.Character or nil
end
local function getHum(p)
    local c = getChar(p)
    return c and c:FindFirstChild("Humanoid") or nil
end
local function getRoot(p)
    local c = getChar(p)
    return c and c:FindFirstChild("HumanoidRootPart") or nil
end
local function getAllPlayers()
    local list = {}
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and getHum(p) and getHum(p).Health > 0 then
            table.insert(list, p)
        end
    end
    return list
end

-- Helper to create labeled toggle
local function makeToggle(parent, name, default, callback)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, -10, 0, 30)
    frame.BackgroundTransparency = 1
    frame.BorderSizePixel = 0
    frame.Parent = parent
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(0, 200, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = name
    label.TextColor3 = Color3.fromRGB(220, 220, 220)
    label.TextSize = 13
    label.Font = Enum.Font.Gotham
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Parent = frame
    
    local toggle = Instance.new("TextButton")
    toggle.Size = UDim2.new(0, 50, 0, 22)
    toggle.Position = UDim2.new(1, -60, 0, 4)
    toggle.BackgroundColor3 = default and Color3.fromRGB(0, 200, 80) or Color3.fromRGB(80, 80, 80)
    toggle.BorderSizePixel = 0
    toggle.Text = default and "ON" or "OFF"
    toggle.TextColor3 = Color3.fromRGB(255, 255, 255)
    toggle.TextSize = 11
    toggle.Font = Enum.Font.GothamBold
    toggle.Parent = frame
    
    local state = default
    toggle.MouseButton1Click:Connect(function()
        state = not state
        toggle.BackgroundColor3 = state and Color3.fromRGB(0, 200, 80) or Color3.fromRGB(80, 80, 80)
        toggle.Text = state and "ON" or "OFF"
        if callback then callback(state) end
    end)
    
    -- Add spacing
    local spacer = Instance.new("Frame")
    spacer.Size = UDim2.new(1, 0, 0, 5)
    spacer.BackgroundTransparency = 1
    spacer.Parent = parent
    
    return frame, toggle, function() return state end
end
