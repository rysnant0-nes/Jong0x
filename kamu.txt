local function GetCharacter()
    return LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
end

local function GetHRP()
    local c = GetCharacter()
    return c and c:FindFirstChild("HumanoidRootPart")
end

local function GetAllRemotes()
    local remotes = {}
    local searchLocations = {
        LocalPlayer.Backpack,
        GetCharacter(),
        ReplicatedStorage
    }
    for _, loc in pairs(searchLocations) do
        if loc then
            for _, v in pairs(loc:GetDescendants()) do
                if v:IsA("RemoteEvent") then
                    table.insert(remotes, v)
                end
            end
        end
    end
    return remotes
end

local function FindRemoteByName(name)
    for _, remote in pairs(GetAllRemotes()) do
        if remote.Name:lower():find(name:lower()) then
            return remote
        end
    end
    return nil
end

local function GetAllPlayers()
    local list = {}
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer then
            table.insert(list, p)
        end
    end
    return list
end

local function GetClosestPlayer(maxDist)
    local hrp = GetHRP()
    if not hrp then return nil end
    local closest, dist = nil, maxDist or 9999
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            local t = p.Character:FindFirstChild("HumanoidRootPart")
            local h = p.Character:FindFirstChildOfClass("Humanoid")
            if t and h and h.Health > 0 then
                local d = (hrp.Position - t.Position).Magnitude
                if d < dist then
                    closest = p
                    dist = d
                end
            end
        end
    end
    return closest, dist
end
